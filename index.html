Javascript : Quiz (1) "Scope"
var a = "satu";​
alert(a);​           // "satu"
 
function alertFunction() {​
    var a;
    alert(a);​       // "undefined"
    a = "dua";​
    alert(a);​       // "dua"
}​
 
alertFunction();

Javascript : Quiz (2) "Chain Command"
var angka = "satu dua tiga";
 
angka.split(' ');      // angka = ["satu","dua","tiga"]
angka.reverse();       // angka = ["tiga", "dua", "satu"]
angka.join(' ');       // angka = "tiga dua satu" 
 
​console.log(angka);
